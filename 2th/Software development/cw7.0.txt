void Compare(int i, int j)
{
	if (i == j)
	{
		printf("equal to");
	}
	else if (i > j)
	{
		printf("more than");
	}
	if (i < j)
	{
		printf("less than");
	}
}

void insert(char str1[p], char str2[p], int i)
{
	char str3[p];
	int m, n, k, l;
	l = strlen(str1);
	for (m = 0; m = i - 1; m++)
	{
		str3[m] = str1[m];
	}
	for (k = 0; str2[k] != '\0'; m++, k++)
	{
		str3[m] = str2[k];
	}
	for (n = l - i - 2; str1[n] != '\0'; m++, n++)
	{
		str3[m] = str1[n];
	}
	str3[m] = '\0';
}
